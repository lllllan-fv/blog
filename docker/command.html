                                    # 服务相关

                                    注意这部分是 Linux 的命令，如果是 Windows 有眼睛的都会启动和停止

                                    • 启动 docker 服务

                                      systemctl start docker
                                      1

                                    • 停止 docker 服务

                                      systemctl stop docker
                                      1

                                    • 重启 docker 服务

                                      systemctl restart docker
                                      1

                                    • 查看 docker 服务状态

                                      systemctl status docker
                                      1

                                    # 镜像相关

                                    # 查看镜像

                                    docker images
docker images -q # 查看所有镜像 id
                                    1
                                    2

                                    # 搜索镜像

                                    docker search <镜像名称>
                                    1

                                    # 拉取镜像

                                    Docker Hubopen in new window 可以搜索镜像查看版本

                                    docker pull <镜像名> # 默认拉取最新版本
docker pull <镜像名>:<版本号>
                                    1
                                    2

                                    # 删除镜像

                                    docker rmi <镜像id> # 删除指定版本的镜像
docker image prune # 删除所有未被 tag 标记和未被容器使用的镜像
docker image prune -a # 删除所有未被容器使用的镜像
                                    1
                                    2

                                    # 容器相关

                                    # 创建并启动容器

                                    • -it 创建的容器一般称为交互式容器，-id 创建的容器一般称为守护式容器
                                      • -i：保持容器运行
                                      • -t：为容器重新分配一个伪输入终端
                                      • -d：以守护（后台）模式运行容器
                                    • --name 为创建的容器命名
                                    docker run <参数> --name <容器名> <镜像名>:<版本号>
                                    1

                                    ::center

                                    :::

                                    # 创建容器

                                    docker create --name <容器名> <镜像名>:<版本号>
                                    1

                                    # 启动容器

                                    docker start <容器名>
                                    1

                                    # 进入容器

                                    docker exec <参数> <容器名> <c>
                                    1

                                    # 停止容器

                                    docker stop <容器名>
                                    1

                                    # 删除容器

                                    docker rm <容器名>
                                    1

                                    # 查看容器

                                    docker ps    # 查看正在运行的容器
docker ps -a # 查看所有容器
                                    1
                                    2
